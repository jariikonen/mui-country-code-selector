const filterOptions = (
  options: CountryType[],
  { inputValue }: { inputValue: string; getOptionLabel: object }
) => {
  const inputVal = inputValue.startsWith('+')
    ? inputValue.substring(1)
    : inputValue;
  return matchSorter<CountryType>(options, inputVal, {
    keys: ['country', 'code', 'iso'],
  });
};
